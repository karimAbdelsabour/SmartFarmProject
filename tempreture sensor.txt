


#include <OneWire.h>
#include <DallasTemperature.h>

#define SENSOR_PIN  17 

OneWire oneWire(SENSOR_PIN);
DallasTemperature DS18B20(&oneWire);

float tempC; 
void setup() {
  Serial.begin(9600); 
  DS18B20.begin();   }

void loop() {
  DS18B20.requestTemperatures();       
  tempC = DS18B20.getTempCByIndex(0);  
 

  Serial.print("Temperature: ");
  Serial.print(tempC);   
  Serial.println("Â°C");
 
  delay(500);
}
